\NewDocumentCommand{\lec}{smm}{
	\expandafter\edef\csname lec@#2@title\endcsname{#3}
	\expandafter\edef\csname lec@#2@part\endcsname{\@curpart}
	\expandafter\edef\csname lec@#2@partnumber\endcsname{\thelec@partnumber}
	\expandafter\edef\csname lec@#2@number\endcsname{\thelec@number}
	\expandafter\edef\csname lec@#2@globalnumber\endcsname{\thelec@globalnumber}
	\IfBooleanF{#1}
	{
		\stepcounter{lec@number}
		\stepcounter{lec@globalnumber}
	}

}

\NewDocumentCommand{\lecpart}{sm}{
	\edef\@curpart{#2}	
	\setcounter{lec@number}{0}

	{
		\stepcounter{lec@partnumber}
	}
	
}
