	\coordinate (sun) at (-2,3);
	\coordinate (eye) at (3,3);
	\coordinate (sun2) at (2,3);
	
	\foreach \t in {45,90,...,360}{

	};
	
	\draw[-Latex, thick] (0,0) -- (0,30mm) node[left]{$\vv n$};
	\draw[-Latex, thick] (0,0) -- ($(0,0)!30mm!(sun)$) node[right]{$\vv l$};
	\draw[-Latex, thick] (0,0) -- ($(0,0)!30mm!(eye)$) node[above]{$\vv v$};
	\draw[-Latex] (0,0) -- ($(0,0)!30mm!(sun2)$) node[above]{$\vv r$};
	
	\draw[-Latex] (0,0) -- ($(0,0)!30mm!($(sun)+(eye)$)$) node[above]{$\vv h$};
	
	
\end{tikzpicture}
